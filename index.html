window.onSpotifyWebPlaybackSDKReady = () => { const token = 'YOUR_ACCESS_TOKEN_HERE'; // Replace with a valid token const playlistId = '4hA72ROeqv56jOEsM5PyV1'; // Your Spotify playlist ID const player = new Spotify.Player({ name: 'VyrizPod Player', getOAuthToken: cb => { cb(token); }, volume: 0.5, }); // Connect the player and handle state changes player.connect().then(success => { if (success) { console.log('Player connected!'); } }); player.addListener('ready', ({ device_id }) => { console.log('Ready with Device ID', device_id); }); player.addListener('not_ready', ({ device_id }) => { console.error('Device ID has gone offline', device_id); }); player.addListener('player_state_changed', state => { if (state) { const currentTrack = state.track_window.current_track; document.getElementById('now-playing').innerText = `Now Playing: ${currentTrack.name} by ${currentTrack.artists.map(artist => artist.name).join(', ')}`; } }); player.addListener('initialization_error', ({ message }) => { console.error('Initialization Error:', message); }); player.addListener('authentication_error', ({ message }) => { console.error('Authentication Error:', message); alert('Authentication error! Please check your Spotify token.'); }); player.addListener('account_error', ({ message }) => { console.error('Account Error:', message); alert('You need a Spotify Premium account to use this feature.'); }); player.addListener('playback_error', ({ message }) => { console.error('Playback Error:', message); alert('Playback error! Please check your Spotify app.'); }); // Fetch playlist and display tracks fetch(`https://api.spotify.com/v1/playlists/${playlistId}/tracks`, { headers: { 'Authorization': `Bearer ${token}` } }) .then(response => response.json()) .then(data => { const playlistDiv = document.getElementById('playlist'); data.items.forEach(item => { const track = item.track; const songElement = document.createElement('div'); songElement.className = 'song'; songElement.innerHTML = ` ${track.name}

By ${track.artists.map(artist => artist.name).join(', ')}

`; songElement.addEventListener('click', () => { player.activateElement().then(() => { player.setVolume(0.5); player.play({ uris: [track.uri] }); }).catch(error => { console.error('Error activating player:', error); alert('Failed to play track. Please make sure your Spotify app is open and running.'); }); }); playlistDiv.appendChild(songElement); }); }) .catch(error => { console.error('Error fetching playlist:', error); alert('Failed to load playlist. Please check your Spotify token or playlist ID.'); }); };
